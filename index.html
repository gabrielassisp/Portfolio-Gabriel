-- Tipo 1: Inativo → Ativo via regex
JOIN cliente_historico ch_novo
  ON substring(ch_novo.historico
    FROM 'servi[çc]o de n[úu]mero \((\d+)\)')
    ::int = cliente_servico.numero_plano
JOIN cliente_historico ch_inativo
  ON ch_inativo.historico ~* 'Inativo para Ativo'
