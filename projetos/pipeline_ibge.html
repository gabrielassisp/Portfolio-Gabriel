# silver_transformer.py — Alinhamento e enriquecimento de séries import pandas as pd   def enrich_series(df: pd.DataFrame, value_col: str = "value") -> pd.DataFrame:     """Calcula média móvel 12m e variação ano a ano (YoY)."""     df = df.copy().sort_values("date")     df["value_12m_avg"] = (         df[value_col].rolling(window=12, min_periods=3).mean().round(4)     )     df["value_yoy_pct"] = (         df[value_col].pct_change(periods=12).mul(100).round(2)     )     return df   def _load_bcb_series(indicator: str) -> pd.DataFrame:     """BCB publica diário — agrega em média mensal via pd.Period.     IBGE publica mensal, PIB trimestral. Tudo alinhado no Silver."""     df["date"] = (pd.to_datetime(df["date"])                         .dt.to_period("M")                         .dt.to_timestamp())     return df.groupby("date")["value"].mean().reset_index()
# gold_builder.py — Índice de Stress Econômico (ISE) def compute_stress_index(     ipca_12m: float | None,     desemprego: float | None,     cambio: float | None, ) -> float | None:     """     ISE = média ponderada normalizada [0–100].      Inflação 12m — peso 40% — ref [0%, 15%]      Desemprego — peso 35% — ref [5%, 20%]      Câmbio R$/USD — peso 25% — ref [3.5, 7.5]     """     if any(v is None for v in [ipca_12m, desemprego, cambio]):         return None     ipca_n = max(0, min(1, ipca_12m / 15.0))     desemp_n = max(0, min(1, (desemprego - 5.0) / 15.0))     cambio_n = max(0, min(1, (cambio - 3.5) / 4.0))     return round((ipca_n * 0.40 + desemp_n * 0.35 + cambio_n * 0.25) * 100, 2)   # Selic real pela fórmula de Fisher (mais precisa que Selic - IPCA) selic_real = ((1 + selic/100) / (1 + ipca_12m/100) - 1) * 100

Contexto do Projeto
Demonstra um pipeline de engenharia de dados completo usando APIs públicas brasileiras reais — IBGE, Banco Central e World Bank. Todas gratuitas e sem cadastro. O pipeline roda offline automaticamente com dados simulados se as APIs estiverem indisponíveis.
Boas Práticas de Desenvolvimento
  • Arquitetura medallion: Bronze preservado, Silver limpo, Gold analítico
  • 42 testes sem banco nem rede — usando fixtures e mocks
  • Validação de qualidade com expectations formais e log auditável
  • conftest.py configura path — sem necessidade de variáveis de ambiente
Decisões Técnicas
  • Bronze nunca é modificado — toda transformação ocorre na Silver
  • BCB publica câmbio diário — pd.Period alinha para mensal sem perda
  • ISE usa pesos e normalização, não média simples entre indicadores
  • Selic real usa fórmula de Fisher, mais precisa que a aproximação linear
